4. Draw a heart

                from turtle import *
                color ('red')
                begin_fill()
                pensize(3)
                left(50)
                forward(133)
                circle(50,200)
                right(140)
                circle(50,200)
                forward(133)
                end_fill()
                done()

5. Draw a infinity

                from turtle import *
                bgcolor('black')
                pencolor('orange')
                pensize(3)
                speed(11)
                right(45)
                for i in range(150):
                   circle(30)
                   if 7 < i <  62:
                      left(5)
                   if(80< i < 133):
                      right(5)
                   if(i < 80):
                     forward(10)
                   else:
                     forward(5)    
                done()

6. Draw a pattern

                    from turtle import *
                    s=Screen()
                    s.title("pattern")
                    pensize(3)
                    speed(0)
                    while True:
                        for i in range(4):
                        forward(50)
                        left(90)
                    right(10)
                    hideturtle()
                    done()
